```cpp
#include <iostream>
#include <thread>
#include <atomic>
#include <mutex>
#include <future>
#include <condition_variable>
#include <atomic>
using namespace std;
int main()
{
    int a = 10;
    int &b = a;
    b += 10;
    cout << a << " " << b << endl;

    //右值引用
    int &&c = move(b);
    c += 10;
    int &&d = 12;
    d += 10;
    cout << a << " " << b << endl;
    cout << c << " " << d << endl;
}
```
